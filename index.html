Python Script: Backup Verifier

import os
from datetime import datetime

source_dir = '/mnt/nas'
backup_dir = '/mnt/san'

def verify_backups():
    for root, _, files in os.walk(source_dir):
        for file in files:
            src_file = os.path.join(root, file)
            rel_path = os.path.relpath(src_file, source_dir)
            backup_file = os.path.join(backup_dir, rel_path)

            if not os.path.exists(backup_file):
                print(f"Missing in backup: {rel_path}")
            else:
                src_size = os.path.getsize(src_file)
                backup_size = os.path.getsize(backup_file)
                if src_size != backup_size:
                    print(f"Size mismatch: {rel_path}")

verify_backups()

Bash: Disk Usage Monitor

#!/bin/bash
THRESHOLD=90

while read line; do
  usage=$(echo $line | awk '{print $5}' | sed 's/%//')
  partition=$(echo $line | awk '{print $6}')
  if [ "$usage" -ge "$THRESHOLD" ]; then
    echo "Warning: $partition usage is at ${usage}%"
  fi
done < <(df -h | grep '^/dev/')

PowerShell: List Installed Apps
